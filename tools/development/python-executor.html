💡 示例代码：刚需演进图谱
import matplotlib.pyplot as plt

# 数据
stages = ["房子（生存）", "数字空间（虚拟身份）", "健康系统（身体保障）", 
          "能源独立（生活自由）", "教育升级（认知资本）", "社群归属（精神家园）"]
values = [1, 2, 3, 4, 5, 6]  # 用来表示演进阶段

# 绘制图表
plt.figure(figsize=(10, 6))
plt.plot(values, stages, marker="o", linestyle="--", color="b")

# 添加注释
for i, stage in enumerate(stages):
    plt.text(values[i]+0.1, i, stage, fontsize=10, va="center")

plt.title("刚需演进图谱：从房子到未来刚需", fontsize=14, fontweight="bold")
plt.xlabel("演进阶段（由低到高）")
plt.ylabel("刚需形态")
plt.grid(True, linestyle="--", alpha=0.6)
plt.tight_layout()

# 显示图表
plt.show()
